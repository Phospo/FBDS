cmake_minimum_required(VERSION 3.31)

project(FBDSimulator LANGUAGES CXX)

cmake_policy(SET CMP0135 NEW)

set(CMAKE_CXX_STANDARD 17)
set(CMAKE_CXX_STANDARD_REQUIRED ON)

include(FetchContent)

FetchContent_Declare(
        googletest
        URL https://github.com/google/googletest/archive/refs/heads/main.zip
        DOWNLOAD_EXTRACT_TIMESTAMP TRUE
)

FetchContent_MakeAvailable(googletest)

enable_testing()

# Wymagany plik dla CDash/CTest gdy nie używany CTestConfig
file(WRITE "${CMAKE_BINARY_DIR}/DartConfiguration.tcl" "# Dart configuration generated by CMake\n")

add_library(FBDSimulatorCore STATIC
    src/model/cinput.cpp
    src/model/coutput.cpp
    src/model/cfunctionblock.cpp
    src/model/cconnection.cpp
    src/model/sdatasample.cpp
    src/model/stimestamp.cpp
    src/odbiornik/codbiornik.cpp
    src/odbiornik/cwritetofile.cpp
    src/engine/csimulatorfbd.cpp
    src/engine/ccontainerfbd.cpp
    src/engine/csimengine.cpp
    src/builder/cfbdbuilder.cpp
    src/generator/csiggen.cpp
    src/generator/cconstval.cpp
    src/generator/cfromcsv.cpp
    src/generator/csingen.cpp
    src/operacje/coperacyjny.cpp
    src/operacje/cgain.cpp
    src/operacje/caddsub.cpp
    src/operacje/cintegral.cpp
)

target_include_directories(FBDSimulatorCore
    PUBLIC
        ${CMAKE_CURRENT_SOURCE_DIR}
        ${CMAKE_CURRENT_SOURCE_DIR}/src/model
        ${CMAKE_CURRENT_SOURCE_DIR}/src/engine
        ${CMAKE_CURRENT_SOURCE_DIR}/src/builder
        ${CMAKE_CURRENT_SOURCE_DIR}/src/generator
        ${CMAKE_CURRENT_SOURCE_DIR}/src/odbiornik
        ${CMAKE_CURRENT_SOURCE_DIR}/src/operacje
)

add_executable(FBDSimulator src/main.cpp)

target_link_libraries(FBDSimulator PRIVATE
    FBDSimulatorCore
)

add_executable(FBDSimulatorTests
    tests/cfunctionblock_tests.cpp
)

target_link_libraries(FBDSimulatorTests PRIVATE
    FBDSimulatorCore
    GTest::gtest_main
)

target_include_directories(FBDSimulatorTests PRIVATE
    ${CMAKE_CURRENT_SOURCE_DIR}
    ${CMAKE_CURRENT_SOURCE_DIR}/src/operacje
)

include(GoogleTest)
gtest_discover_tests(FBDSimulatorTests)
